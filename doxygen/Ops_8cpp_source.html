44 /// This class defines the interface for handling inlining with standard
45 /// operations.
46 struct StdInlinerInterface : public DialectInlinerInterface {

51  //===--------------------------------------------------------------------===//
52 
53  /// All call operations within standard ops can be inlined.
54  bool isLegalToInline(Operation *call, Operation *callable,
55  bool wouldBeCloned) const final {
56  return true;
57  }
58 
59  /// All operations within standard ops can be inlined.
60  bool isLegalToInline(Operation *, Region *, bool,
61  BlockAndValueMapping &) const final {
62  return true;
63  }

65  //===--------------------------------------------------------------------===//
66  // Transformation Hooks
67  //===--------------------------------------------------------------------===//
68 
69  /// Handle the given inlined terminator by replacing it with a new operation
70  /// as necessary.
71  void handleTerminator(Operation *op, Block *newDest) const final {
72  // Only "std.return" needs to be handled here.
73  auto returnOp = dyn_cast<ReturnOp>(op);
74  if (!returnOp)
75  return;
76 
77  // Replace the return with a branch to the dest.
78  OpBuilder builder(op);
79  builder.create<cf::BranchOp>(op->getLoc(), newDest, returnOp.getOperands());
80  op->erase();
81  }
82 
83  /// Handle the given inlined terminator by replacing it with a new operation
84  /// as necessary.
85  void handleTerminator(Operation *op,
86  ArrayRef<Value> valuesToRepl) const final {
87  // Only "std.return" needs to be handled here.
88  auto returnOp = cast<ReturnOp>(op);
89 
90  // Replace the values directly with the return operands.
91  assert(returnOp.getNumOperands() == valuesToRepl.size());
92  for (const auto &it : llvm::enumerate(returnOp.getOperands()))
93  valuesToRepl[it.index()].replaceAllUsesWith(it.value());
94  }
95 };
96 } // namespace

127 LogicalResult CallOp::verifySymbolUses(SymbolTableCollection &symbolTable) {
128  // Check that the callee attribute was specified.
129  auto fnAttr = (*this)->getAttrOfType<FlatSymbolRefAttr>("callee");
130  if (!fnAttr)
131  return emitOpError("requires a 'callee' symbol reference attribute");
132  FuncOp fn = symbolTable.lookupNearestSymbolFrom<FuncOp>(*this, fnAttr);
133  if (!fn)
134  return emitOpError() << "'" << fnAttr.getValue()
135  << "' does not reference a valid function";
136 
137  // Verify that the operand and result types match the callee.
138  auto fnType = fn.getType();
139  if (fnType.getNumInputs() != getNumOperands())
140  return emitOpError("incorrect number of operands for callee");
141 
142  for (unsigned i = 0, e = fnType.getNumInputs(); i != e; ++i)
143  if (getOperand(i).getType() != fnType.getInput(i))
144  return emitOpError("operand type mismatch: expected operand type ")
145  << fnType.getInput(i) << ", but provided "
146  << getOperand(i).getType() << " for operand number " << i;
147 
148  if (fnType.getNumResults() != getNumResults())
149  return emitOpError("incorrect number of results for callee");
150 
151  for (unsigned i = 0, e = fnType.getNumResults(); i != e; ++i)
152  if (getResult(i).getType() != fnType.getResult(i)) {
153  auto diag = emitOpError("result type mismatch at index ") << i;
154  diag.attachNote() << " op result types: " << getResultTypes();
155  diag.attachNote() << "function result types: " << fnType.getResults();
156  return diag;
157  }
158 
159  return success();
160 }

170 /// Fold indirect calls that have a constant function as the callee operand.
171 LogicalResult CallIndirectOp::canonicalize(CallIndirectOp indirectCall,
172  PatternRewriter &rewriter) {
173  // Check that the callee is a constant callee.
174  SymbolRefAttr calledFn;
175  if (!matchPattern(indirectCall.getCallee(), m_Constant(&calledFn)))
176  return failure();
177 
178  // Replace with a direct call.
179  rewriter.replaceOpWithNewOp<CallOp>(indirectCall, calledFn,
180  indirectCall.getResultTypes(),
181  indirectCall.getArgOperands());
182  return success();
183 }

189 LogicalResult ConstantOp::verify() {
190  StringRef fnName = getValue();
191  Type type = getType();
192 
193  // Try to find the referenced function.
194  auto fn = (*this)->getParentOfType<ModuleOp>().lookupSymbol<FuncOp>(fnName);
195  if (!fn)
196  return emitOpError() << "reference to undefined function '" << fnName
197  << "'";
198 
199  // Check that the referenced function has the correct type.
200  if (fn.getType() != type)
201  return emitOpError("reference to function with mismatched type");
202 
203  return success();
204 }

224 LogicalResult ReturnOp::verify() {
225  auto function = cast<FuncOp>((*this)->getParentOp());
226 
227  // The operand number and types must match the function signature.
228  const auto &results = function.getType().getResults();
229  if (getNumOperands() != results.size())
230  return emitOpError("has ")
231  << getNumOperands() << " operands, but enclosing function (@"
232  << function.getName() << ") returns " << results.size();
233 
234  for (unsigned i = 0, e = results.size(); i != e; ++i)
235  if (getOperand(i).getType() != results[i])
236  return emitError() << "type of return operand " << i << " ("
237  << getOperand(i).getType()
238  << ") doesn't match function result type ("
239  << results[i] << ")"
240  << " in function @" << function.getName();
241 
242  return success();
243 }
