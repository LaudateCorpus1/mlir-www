23 /// PresburgerSpace is a tuple of identifiers with information about what kind
24 /// they correspond to. The identifiers can be split into three types:
25 ///
26 /// Dimension: Ordinary variables over which the space is represented.
27 ///
28 /// Symbol: Symbol identifiers correspond to fixed but unknown values.
29 /// Mathematically, a space with symbolic identifiers is like a
30 /// family of spaces indexed by the symbolic identifiers.
31 ///
32 /// Local: Local identifiers correspond to existentially quantified variables.
33 ///
34 /// PresburgerSpace only supports identifiers of kind Dimension and Symbol.
35 class PresburgerSpace {
36  friend PresburgerLocalSpace;
37 
38 public:
39  /// Kind of identifier (column).
40  enum IdKind { Dimension, Symbol, Local };
41 
42  PresburgerSpace(unsigned numDims, unsigned numSymbols)
43  : numDims(numDims), numSymbols(numSymbols), numLocals(0) {}
44 
45  virtual ~PresburgerSpace() = default;
46 
47  unsigned getNumIds() const { return numDims + numSymbols + numLocals; }
48  unsigned getNumDimIds() const { return numDims; }
49  unsigned getNumSymbolIds() const { return numSymbols; }
50  unsigned getNumDimAndSymbolIds() const { return numDims + numSymbols; }
51 
52  /// Get the number of ids of the specified kind.
53  unsigned getNumIdKind(IdKind kind) const;
54 
55  /// Return the index at which the specified kind of id starts.
56  unsigned getIdKindOffset(IdKind kind) const;
57 
58  /// Return the index at Which the specified kind of id ends.
59  unsigned getIdKindEnd(IdKind kind) const;
60 
61  /// Get the number of elements of the specified kind in the range
62  /// [idStart, idLimit).
63  unsigned getIdKindOverlap(IdKind kind, unsigned idStart,
64  unsigned idLimit) const;
65 
66  /// Insert `num` identifiers of the specified kind at position `pos`.
67  /// Positions are relative to the kind of identifier. Return the absolute
68  /// column position (i.e., not relative to the kind of identifier) of the
69  /// first added identifier.
70  virtual unsigned insertId(IdKind kind, unsigned pos, unsigned num = 1);
71 
72  /// Removes identifiers in the column range [idStart, idLimit).
73  virtual void removeIdRange(unsigned idStart, unsigned idLimit);
74 
75  /// Changes the partition between dimensions and symbols. Depending on the new
76  /// symbol count, either a chunk of dimensional identifiers immediately before
77  /// the split become symbols, or some of the symbols immediately after the
78  /// split become dimensions.
79  void setDimSymbolSeparation(unsigned newSymbolCount);
80 
81 private:
82  PresburgerSpace(unsigned numDims, unsigned numSymbols, unsigned numLocals)
83  : numDims(numDims), numSymbols(numSymbols), numLocals(numLocals) {}
84 
85  /// Number of identifiers corresponding to real dimensions.
86  unsigned numDims;
87 
88  /// Number of identifiers corresponding to symbols (unknown but constant for
89  /// analysis).
90  unsigned numSymbols;
91 
92  /// Total number of identifiers.
93  unsigned numLocals;
94 };
