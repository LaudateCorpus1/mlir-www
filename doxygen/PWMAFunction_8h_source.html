10 // defined on a domain that is a union of IntegerPolyhedrons, and on each domain

24 /// This class represents a multi-affine function whose domain is given by an
25 /// IntegerPolyhedron. This can be thought of as an IntegerPolyhedron with a
26 /// tuple of integer values attached to every point in the polyhedron, with the
27 /// value of each element of the tuple given by an affine expression in the ids
28 /// of the polyhedron. For example we could have the domain
29 ///
30 /// (x, y) : (x >= 5, y >= x)
31 ///
32 /// and a tuple of three integers defined at every point in the polyhedron:
33 ///
34 /// (x, y) -> (x + 2, 2*x - 3y + 5, 2*x + y).
35 ///
36 /// In this way every point in the polyhedron has a tuple of integers associated
37 /// with it. If the integer polyhedron has local ids, then the output
38 /// expressions can use them as well. The output expressions are represented as
39 /// a matrix with one row for every element in the output vector one column for
40 /// each id, and an extra column at the end for the constant term.
41 ///
42 /// Checking equality of two such functions is supported, as well as finding the
43 /// value of the function at a specified point. Note that local ids in the
44 /// domain are not yet supported for finding the value at a point.
45 class MultiAffineFunction : protected IntegerPolyhedron {
46 public:
47  /// We use protected inheritance to avoid inheriting the whole public
48  /// interface of IntegerPolyhedron. These using declarations explicitly make
49  /// only the relevant functions part of the public interface.
50  using IntegerPolyhedron::getNumDimAndSymbolIds;
51  using IntegerPolyhedron::getNumDimIds;
52  using IntegerPolyhedron::getNumIds;
53  using IntegerPolyhedron::getNumLocalIds;
54  using IntegerPolyhedron::getNumSymbolIds;

57  : IntegerPolyhedron(domain), output(output) {}

59  unsigned numSymbols = 0, unsigned numLocals = 0)
60  : IntegerPolyhedron(numDims, numSymbols, numLocals), output(output) {}

122 /// This class represents a piece-wise MultiAffineFunction. This can be thought
123 /// of as a list of MultiAffineFunction with disjoint domains, with each having
124 /// their own affine expressions for their output tuples. For example, we could
125 /// have a function with two input variables (x, y), defined as
126 ///
127 /// f(x, y) = (2*x + y, y - 4) if x >= 0, y >= 0
128 /// = (-2*x + y, y + 4) if x < 0, y < 0
129 /// = (4, 1) if x < 0, y >= 0
130 ///
131 /// Note that the domains all have to be *disjoint*. Otherwise, the behaviour of
132 /// this class is undefined. The domains need not cover all possible points;
133 /// this represents a partial function and so could be undefined at some points.
134 ///
135 /// As in PresburgerSets, the input ids are partitioned into dimension ids and
136 /// symbolic ids.
137 ///
138 /// Support is provided to compare equality of two such functions as well as
139 /// finding the value of the function at a point. Note that local ids in the
140 /// piece are not supported for the latter.
141 class PWMAFunction : PresburgerSpace {
142 public:
143  PWMAFunction(unsigned numDims, unsigned numSymbols, unsigned numOutputs)
144  : PresburgerSpace(numDims, numSymbols), numOutputs(numOutputs) {
145  assert(numOutputs >= 1 && "The function must output something!");
146  }
147 
148  void addPiece(const MultiAffineFunction &piece);
149  void addPiece(const IntegerPolyhedron &domain, const Matrix &output);
150 
151  const MultiAffineFunction &getPiece(unsigned i) const { return pieces[i]; }
152  unsigned getNumPieces() const { return pieces.size(); }
153  unsigned getNumOutputs() const { return numOutputs; }
154  unsigned getNumInputs() const { return getNumIds(); }
155  MultiAffineFunction &getPiece(unsigned i) { return pieces[i]; }
156 
157  /// Return the domain of this piece-wise MultiAffineFunction. This is the
158  /// union of the domains of all the pieces.
159  PresburgerSet getDomain() const;
160 
161  /// Check whether the `this` and the given function have compatible
162  /// dimensions, i.e., the same number of dimension inputs, symbol inputs, and
163  /// outputs.
164  bool hasCompatibleDimensions(const MultiAffineFunction &f) const;
165  bool hasCompatibleDimensions(const PWMAFunction &f) const;
166 
167  /// Return the value at the specified point and an empty optional if the
168  /// point does not lie in the domain.
169  ///
170  /// Note: domains with local ids are not yet supported, and will assert-fail.
171  Optional<SmallVector<int64_t, 8>> valueAt(ArrayRef<int64_t> point) const;
172 
173  /// Return whether `this` and `other` are equal as PWMAFunctions, i.e. whether
174  /// they have the same dimensions, the same domain and they take the same
175  /// value at every point in the domain.
176  bool isEqual(const PWMAFunction &other) const;
177 
178  void print(raw_ostream &os) const;
179  void dump() const;
180 
181 private:
182  /// The list of pieces in this piece-wise MultiAffineFunction.
183  SmallVector<MultiAffineFunction, 4> pieces;
184 
185  /// The number of output ids.
186  unsigned numOutputs;
187 };
