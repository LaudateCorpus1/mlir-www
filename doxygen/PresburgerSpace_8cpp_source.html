39 unsigned PresburgerSpace::getIdKindOverlap(IdKind kind, unsigned idStart,
40  unsigned idLimit) const {
41  unsigned idRangeStart = getIdKindOffset(kind);
42  unsigned idRangeEnd = getIdKindEnd(kind);
43 
44  // Compute number of elements in intersection of the ranges [idStart, idLimit)
45  // and [idRangeStart, idRangeEnd).
46  unsigned overlapStart = std::max(idStart, idRangeStart);
47  unsigned overlapEnd = std::min(idLimit, idRangeEnd);
48 
49  if (overlapStart > overlapEnd)
50  return 0;
51  return overlapEnd - overlapStart;
52 }

70 void PresburgerSpace::removeIdRange(unsigned idStart, unsigned idLimit) {
71  assert(idLimit <= getNumIds() && "invalid id limit");
72 
73  if (idStart >= idLimit)
74  return;
75 
76  // We are going to be removing one or more identifiers from the range.
77  assert(idStart < getNumIds() && "invalid idStart position");
78 
79  // Update members numDims, numSymbols and numIds.
80  unsigned numDimsEliminated =
81  getIdKindOverlap(IdKind::Dimension, idStart, idLimit);
82  unsigned numSymbolsEliminated =
83  getIdKindOverlap(IdKind::Symbol, idStart, idLimit);
84 
85  numDims -= numDimsEliminated;
86  numSymbols -= numSymbolsEliminated;
